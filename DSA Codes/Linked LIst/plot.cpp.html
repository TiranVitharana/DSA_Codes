#include <stdio.h>
#include <string.h>
#include <iostream>
using namespace std;

struct Node {
    int data;
    struct Node *next;
};

struct Node *head = nullptr;

void insert(int data) {

    struct Node* new_node = new Node;
    new_node->data = data;
    new_node->next = head;
    head = new_node;
}
void display() {
    struct Node* ptr;
    ptr = head;
    while (ptr != nullptr) {
        cout << ptr->data << " ";
        ptr = ptr->next;
    }
    cout << "\n";
}

void search (int key) {
    int position =1;
    bool found = false;
    struct Node* ptr;
    ptr = head;
    while (true) {
        if (ptr == nullptr)  {
            cout << "Not Found";
            return;
        }
        if (ptr->data == key) {
            cout << position;
            return;
        }
        else {
            ptr = ptr->next;
            position+=1;
        }
    }
}
void insertPos(int key, int pos) {


    struct Node* new_node;
    new_node->data = key;

    if (pos == 1) {
        new_node->next = head;
        head = new_node;
        return;
    }

    int curr_position =1;
    struct Node* prev;
    struct Node* ptr;
    prev = nullptr;
    ptr = head;
    while (curr_position<pos && ptr!= nullptr) {
        prev = ptr;
        ptr = ptr ->next;
        curr_position++;

    }
    new_node->next = ptr;
    prev->next = new_node;
}

int main() {
    insert(5);

    insert(0);
    display();
    insertPos(3,1);
    display();
}
